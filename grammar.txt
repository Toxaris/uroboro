{repetition; comma or space separated}

f ::= function names
x ::= variable names
c ::= constructor names
s ::= selector names
D ::= data type names
C ::= codata type names

t ::= D | C     # Type

e ::= x         # Expression
    | f({e})
    | c({e})
    | e.s({e})

p ::= x | c({p})    # Pattern

d ::= data D where      # Definition
         {c({t}): t}
    | codata C where
         {C.s({t}): t}
    | function f({t}): t where
         {f({p}){.s({p})} = e}

P ::= {d}       # Programm
